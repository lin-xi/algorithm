  • 跟插入排序的主要区别是，在有序的数组中查找，二分查找更好
    
    function bubbleSort(arr: Array): [Array, number] {
      let cnt: number = 0
      for (let i: number = arr.length - 1; i > 0; i--) {
        cnt++;
        for (let j: number = 0; j < i; j++) {
          cnt++;
          if (arr[j] > arr[j + 1]) {
            [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]]
          }
        }
      }
      return [arr, cnt]
    }
